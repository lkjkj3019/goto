----------회원 테이블------------
CREATE TABLE MEMBERS(          --회원
    MCODE   NVARCHAR2(5),      --회원코드
    MID NVARCHAR2(20),         --회원아이디
    MPASSWORD   NVARCHAR2(20), --회원비번
    MNAME   NVARCHAR2(5),      --회원이름
    MPHONE NVARCHAR2(13),      --회원폰번
    MBIRTH  DATE,              --회원생년월일
    MEMAIL  NVARCHAR2(20)      --회원이메일
);
---회원 제약조건---
ALTER TABLE MEMBERS
MODIFY MID NOT NULL
ADD CONSTRAINT MID_UQ UNIQUE(MID)
MODIFY MPASSWORD NOT NULL
MODIFY MNAME NOT NULL
MODIFY MPHONE NOT NULL
MODIFY MBIRTH NOT NULL
MODIFY MEMAIL NOT NULL
ADD CONSTRAINT MCODE_PK PRIMARY KEY(MCODE);
